### `market/market_data.proto`

Defines the market data service with **server streaming**.

```protobuf
syntax = "proto3";

package market;

import "common/types.proto";

option java_package = "com.example.trading.proto.market";
option java_multiple_files = true;

// ─── Messages ────────────────────────────────────────

// A tradeable instrument (stock, crypto, forex)
message Instrument {
  string symbol = 1;           // e.g., "AAPL", "BTC-USD"
  string name = 2;             // e.g., "Apple Inc."
  InstrumentType type = 3;
}

enum InstrumentType {
  INSTRUMENT_TYPE_UNSPECIFIED = 0;
  INSTRUMENT_TYPE_STOCK = 1;
  INSTRUMENT_TYPE_CRYPTO = 2;
  INSTRUMENT_TYPE_FOREX = 3;
  INSTRUMENT_TYPE_COMMODITY = 4;
}

// A single price update
message PriceTick {
  string symbol = 1;
  common.Decimal bid_price = 2;   // price buyers will pay
  common.Decimal ask_price = 3;   // price sellers want
  common.Decimal last_price = 4;  // last traded price
  int64 volume = 5;               // shares/units traded
  common.Timestamp timestamp = 6;
  PriceDirection direction = 7;
}

enum PriceDirection {
  PRICE_DIRECTION_UNSPECIFIED = 0;
  PRICE_DIRECTION_UP = 1;
  PRICE_DIRECTION_DOWN = 2;
  PRICE_DIRECTION_UNCHANGED = 3;
}

// ─── Request / Response ──────────────────────────────

message SubscribeRequest {
  repeated string symbols = 1;     // subscribe to these symbols
  int32 throttle_ms = 2;           // min interval between ticks (0 = real-time)
}

message MarketSummaryRequest {
  string symbol = 1;
}

message MarketSummary {
  Instrument instrument = 1;
  common.Decimal open_price = 2;
  common.Decimal high_price = 3;
  common.Decimal low_price = 4;
  common.Decimal close_price = 5;
  int64 total_volume = 6;
  common.Timestamp as_of = 7;
}

// ─── Service ─────────────────────────────────────────

service MarketDataService {
  // Unary — get a market summary for a single symbol
  rpc GetMarketSummary(MarketSummaryRequest) returns (MarketSummary);

  // Server Streaming — subscribe to live price ticks
  rpc StreamPrices(SubscribeRequest) returns (stream PriceTick);
}
```

### `trading/trading.proto`

Defines the trading service with **unary**, **client streaming**, and **bidirectional streaming**.

```protobuf
syntax = "proto3";

package trading;

import "common/types.proto";
import "market/market_data.proto";  // cross-service import for Instrument

option java_package = "com.example.trading.proto.trading";
option java_multiple_files = true;

// ─── Core Messages ───────────────────────────────────

message TradeOrder {
  string order_id = 1;
  string client_id = 2;
  market.Instrument instrument = 3;   // ← from market_data.proto
  OrderSide side = 4;
  OrderType type = 5;
  common.Decimal quantity = 6;        // ← from common/types.proto
  common.Decimal limit_price = 7;     // only for LIMIT orders
  common.Timestamp submitted_at = 8;
  common.Metadata metadata = 9;
}

enum OrderSide {
  ORDER_SIDE_UNSPECIFIED = 0;
  ORDER_SIDE_BUY = 1;
  ORDER_SIDE_SELL = 2;
}

enum OrderType {
  ORDER_TYPE_UNSPECIFIED = 0;
  ORDER_TYPE_MARKET = 1;    // execute at current market price
  ORDER_TYPE_LIMIT = 2;     // execute only at specified price or better
  ORDER_TYPE_STOP = 3;      // trigger when price hits threshold
}

// Sent by server when an order is filled / partially filled / rejected
message ExecutionReport {
  string execution_id = 1;
  string order_id = 2;          // links back to the original order
  string client_id = 3;
  ExecutionStatus status = 4;
  common.Decimal filled_quantity = 5;
  common.Decimal fill_price = 6;
  common.Decimal remaining_quantity = 7;
  common.Timestamp executed_at = 8;
  string message = 9;           // human-readable status message
}

enum ExecutionStatus {
  EXECUTION_STATUS_UNSPECIFIED = 0;
  EXECUTION_STATUS_ACCEPTED = 1;       // order acknowledged
  EXECUTION_STATUS_PARTIAL_FILL = 2;   // partially executed
  EXECUTION_STATUS_FILLED = 3;         // fully executed
  EXECUTION_STATUS_REJECTED = 4;       // order rejected
  EXECUTION_STATUS_CANCELLED = 5;      // order cancelled
}

// ─── Request / Response Messages ─────────────────────

// Unary
message PlaceOrderRequest {
  TradeOrder order = 1;
}

// Client streaming response — summary after batch
message BatchOrderResponse {
  int32 total_received = 1;
  int32 accepted = 2;
  int32 rejected = 3;
  repeated ExecutionReport reports = 4;
  common.Timestamp processed_at = 5;
}

// Bidirectional — client sends commands, server sends execution reports
message TradeCommand {
  string command_id = 1;
  string client_id = 2;
  common.Timestamp timestamp = 3;

  oneof action {
    TradeOrder place_order = 10;
    CancelOrderRequest cancel_order = 11;
    ModifyOrderRequest modify_order = 12;
    HeartbeatRequest heartbeat = 13;
  }
}

message CancelOrderRequest {
  string order_id = 1;
  string reason = 2;
}

message ModifyOrderRequest {
  string order_id = 1;
  common.Decimal new_quantity = 2;
  common.Decimal new_limit_price = 3;
}

message HeartbeatRequest {
  // empty — just keeps the connection alive
}

// Server response in the bidi stream
message TradeEvent {
  string event_id = 1;
  common.Timestamp timestamp = 2;

  oneof payload {
    ExecutionReport execution = 10;
    TradeError error = 11;
    HeartbeatResponse heartbeat = 12;
    SystemAlert alert = 13;
  }
}

message TradeError {
  string command_id = 1;   // links to the command that caused the error
  int32 error_code = 2;
  string message = 3;
}

message HeartbeatResponse {
  int64 server_time_ms = 1;
  int32 active_orders = 2;
}

message SystemAlert {
  AlertLevel level = 1;
  string message = 2;
}

enum AlertLevel {
  ALERT_LEVEL_UNSPECIFIED = 0;
  ALERT_LEVEL_INFO = 1;
  ALERT_LEVEL_WARNING = 2;
  ALERT_LEVEL_CRITICAL = 3;
}

// ─── Service Definition ──────────────────────────────

service TradingService {
  // Unary — place a single order
  rpc PlaceOrder(PlaceOrderRequest) returns (ExecutionReport);

  // Client Streaming — submit a batch of orders, get summary
  rpc BatchPlaceOrders(stream TradeOrder) returns (BatchOrderResponse);

  // ⭐ Bidirectional Streaming — live trading session
  // Client sends TradeCommands (orders, cancels, modifies, heartbeats)
  // Server sends TradeEvents (executions, errors, heartbeats, alerts)
  rpc LiveTrading(stream TradeCommand) returns (stream TradeEvent);
}
```

```java
package com.example.trading.server;

import io.grpc.Server;
import io.grpc.ServerBuilder;

import java.io.IOException;
import java.util.logging.Logger;

public class TradingServer {

    private static final Logger logger = Logger.getLogger(TradingServer.class.getName());
    private static final int PORT = 6565;

    public static void main(String[] args) throws IOException, InterruptedException {
        Server server = ServerBuilder.forPort(PORT)
                .addService(new MarketDataServiceImpl())
                .addService(new TradingServiceImpl())
                .build()
                .start();

        logger.info("═══════════════════════════════════════════");
        logger.info("  Trading Platform gRPC Server");
        logger.info("  Port: " + PORT);
        logger.info("  Services:");
        logger.info("    ✓ MarketDataService (unary + server streaming)");
        logger.info("    ✓ TradingService    (unary + client + bidi streaming)");
        logger.info("═══════════════════════════════════════════");

        Runtime.getRuntime().addShutdownHook(new Thread(() -> {
            logger.info("Shutting down Trading Server...");
            server.shutdown();
        }));

        server.awaitTermination();
    }
}
```

### `MarketDataServiceImpl.java`

Implements **unary** (`GetMarketSummary`) and **server streaming** (`StreamPrices`).

```java
package com.example.trading.server;

import com.example.trading.proto.common.Decimal;
import com.example.trading.proto.common.Timestamp;
import com.example.trading.proto.market.*;
import io.grpc.stub.StreamObserver;

import java.util.List;
import java.util.Random;
import java.util.logging.Logger;

public class MarketDataServiceImpl extends MarketDataServiceGrpc.MarketDataServiceImplBase {

    private static final Logger logger = Logger.getLogger(MarketDataServiceImpl.class.getName());
    private final Random random = new Random();

    // ─── Unary: GetMarketSummary ─────────────────────

    @Override
    public void getMarketSummary(MarketSummaryRequest request,
                                  StreamObserver<MarketSummary> responseObserver) {

        String symbol = request.getSymbol();
        logger.info("[MarketData] GetMarketSummary for: " + symbol);

        MarketSummary summary = MarketSummary.newBuilder()
                .setInstrument(Instrument.newBuilder()
                        .setSymbol(symbol)
                        .setName(getInstrumentName(symbol))
                        .setType(InstrumentType.INSTRUMENT_TYPE_STOCK)
                        .build())
                .setOpenPrice(makeDecimal(15000, 2))
                .setHighPrice(makeDecimal(15525, 2))
                .setLowPrice(makeDecimal(14890, 2))
                .setClosePrice(makeDecimal(15200, 2))
                .setTotalVolume(2_500_000)
                .setAsOf(now())
                .build();

        responseObserver.onNext(summary);
        responseObserver.onCompleted();
    }

    // ─── Server Streaming: StreamPrices ──────────────

    @Override
    public void streamPrices(SubscribeRequest request,
                             StreamObserver<PriceTick> responseObserver) {

        List<String> symbols = request.getSymbolsList();
        int throttleMs = request.getThrottleMs() > 0 ? request.getThrottleMs() : 500;

        logger.info("[MarketData] StreamPrices started — symbols: "
                + symbols + ", throttle: " + throttleMs + "ms");

        // Simulate a live price feed — send 20 ticks per symbol
        try {
            for (int tick = 0; tick < 20; tick++) {
                for (String symbol : symbols) {
                    int basePrice = getBasePrice(symbol);
                    int variation = random.nextInt(200) - 100; // ±1.00

                    PriceTick priceTick = PriceTick.newBuilder()
                            .setSymbol(symbol)
                            .setBidPrice(makeDecimal(basePrice + variation - 5, 2))
                            .setAskPrice(makeDecimal(basePrice + variation + 5, 2))
                            .setLastPrice(makeDecimal(basePrice + variation, 2))
                            .setVolume(random.nextInt(10000) + 100)
                            .setTimestamp(now())
                            .setDirection(variation > 0
                                    ? PriceDirection.PRICE_DIRECTION_UP
                                    : PriceDirection.PRICE_DIRECTION_DOWN)
                            .build();

                    responseObserver.onNext(priceTick);
                }
                Thread.sleep(throttleMs);
            }
        } catch (InterruptedException e) {
            Thread.currentThread().interrupt();
        }

        logger.info("[MarketData] StreamPrices completed for: " + symbols);
        responseObserver.onCompleted();
    }

    // ─── Helpers ─────────────────────────────────────

    private Decimal makeDecimal(long unscaled, int scale) {
        return Decimal.newBuilder()
                .setUnscaledValue(unscaled)
                .setScale(scale)
                .build();
    }

    private Timestamp now() {
        long millis = System.currentTimeMillis();
        return Timestamp.newBuilder()
                .setSeconds(millis / 1000)
                .setNanos((int) ((millis % 1000) * 1_000_000))
                .build();
    }

    private int getBasePrice(String symbol) {
        switch (symbol) {
            case "AAPL":    return 17850;  // $178.50
            case "GOOGL":   return 14200;  // $142.00
            case "MSFT":    return 38000;  // $380.00
            case "BTC-USD": return 6800000; // $68,000.00
            default:        return 10000;
        }
    }

    private String getInstrumentName(String symbol) {
        switch (symbol) {
            case "AAPL":    return "Apple Inc.";
            case "GOOGL":   return "Alphabet Inc.";
            case "MSFT":    return "Microsoft Corp.";
            case "BTC-USD": return "Bitcoin";
            default:        return symbol;
        }
    }
}
```

### `TradingServiceImpl.java`

Implements **unary** (`PlaceOrder`), **client streaming** (`BatchPlaceOrders`), and **⭐ bidirectional streaming** (`LiveTrading`).

```java
package com.example.trading.server;

import com.example.trading.proto.common.Decimal;
import com.example.trading.proto.common.Timestamp;
import com.example.trading.proto.trading.*;
import io.grpc.stub.StreamObserver;

import java.util.*;
import java.util.concurrent.ConcurrentHashMap;
import java.util.concurrent.atomic.AtomicInteger;
import java.util.logging.Logger;

public class TradingServiceImpl extends TradingServiceGrpc.TradingServiceImplBase {

    private static final Logger logger = Logger.getLogger(TradingServiceImpl.class.getName());
    private final AtomicInteger executionCounter = new AtomicInteger(0);
    private final Random random = new Random();

    // ═══════════════════════════════════════════════════
    // 1. UNARY — PlaceOrder
    // ═══════════════════════════════════════════════════

    @Override
    public void placeOrder(PlaceOrderRequest request,
                           StreamObserver<ExecutionReport> responseObserver) {

        TradeOrder order = request.getOrder();
        logger.info("[Trading] PlaceOrder: " + order.getOrderId()
                + " " + order.getSide() + " " + order.getInstrument().getSymbol());

        // Simulate order execution
        ExecutionReport report = executeOrder(order);

        responseObserver.onNext(report);
        responseObserver.onCompleted();
    }

    // ═══════════════════════════════════════════════════
    // 2. CLIENT STREAMING — BatchPlaceOrders
    // ═══════════════════════════════════════════════════

    @Override
    public StreamObserver<TradeOrder> batchPlaceOrders(
            StreamObserver<BatchOrderResponse> responseObserver) {

        logger.info("[Trading] BatchPlaceOrders stream opened");

        // Return a StreamObserver that accumulates incoming orders
        return new StreamObserver<TradeOrder>() {

            private final List<TradeOrder> collectedOrders = new ArrayList<>();

            @Override
            public void onNext(TradeOrder order) {
                // Called for each order the client sends
                logger.info("[Trading] Batch received order: " + order.getOrderId());
                collectedOrders.add(order);
            }

            @Override
            public void onCompleted() {
                // Client finished sending — process all orders and respond
                logger.info("[Trading] Batch stream completed — processing "
                        + collectedOrders.size() + " orders");

                int accepted = 0;
                int rejected = 0;
                List<ExecutionReport> reports = new ArrayList<>();

                for (TradeOrder order : collectedOrders) {
                    ExecutionReport report = executeOrder(order);
                    reports.add(report);

                    if (report.getStatus() == ExecutionStatus.EXECUTION_STATUS_REJECTED) {
                        rejected++;
                    } else {
                        accepted++;
                    }
                }

                BatchOrderResponse response = BatchOrderResponse.newBuilder()
                        .setTotalReceived(collectedOrders.size())
                        .setAccepted(accepted)
                        .setRejected(rejected)
                        .addAllReports(reports)
                        .setProcessedAt(now())
                        .build();

                responseObserver.onNext(response);
                responseObserver.onCompleted();
            }

            @Override
            public void onError(Throwable t) {
                logger.warning("[Trading] Batch stream error: " + t.getMessage());
            }
        };
    }

    // ═══════════════════════════════════════════════════
    // 3. ⭐ BIDIRECTIONAL STREAMING — LiveTrading
    // ═══════════════════════════════════════════════════

    @Override
    public StreamObserver<TradeCommand> liveTrading(
            StreamObserver<TradeEvent> responseObserver) {

        logger.info("[Trading] ⭐ LiveTrading bidi stream opened");

        // Track active orders for this session
        Map<String, TradeOrder> activeOrders = new ConcurrentHashMap<>();

        // Send an initial system alert to welcome the client
        responseObserver.onNext(TradeEvent.newBuilder()
                .setEventId(generateEventId())
                .setTimestamp(now())
                .setAlert(SystemAlert.newBuilder()
                        .setLevel(AlertLevel.ALERT_LEVEL_INFO)
                        .setMessage("Connected to Trading Engine v2.0. Session active.")
                        .build())
                .build());

        // Return a StreamObserver that handles each incoming command
        return new StreamObserver<TradeCommand>() {

            @Override
            public void onNext(TradeCommand command) {
                // Called each time the client sends a command
                // We process it and may send one or more TradeEvents back

                logger.info("[Trading] ⭐ Received command: " + command.getCommandId()
                        + " type=" + command.getActionCase());

                switch (command.getActionCase()) {

                    case PLACE_ORDER:
                        handlePlaceOrder(command, responseObserver, activeOrders);
                        break;

                    case CANCEL_ORDER:
                        handleCancelOrder(command, responseObserver, activeOrders);
                        break;

                    case MODIFY_ORDER:
                        handleModifyOrder(command, responseObserver, activeOrders);
                        break;

                    case HEARTBEAT:
                        handleHeartbeat(command, responseObserver, activeOrders);
                        break;

                    default:
                        // Unknown command type — send error back
                        responseObserver.onNext(TradeEvent.newBuilder()
                                .setEventId(generateEventId())
                                .setTimestamp(now())
                                .setError(TradeError.newBuilder()
                                        .setCommandId(command.getCommandId())
                                        .setErrorCode(400)
                                        .setMessage("Unknown command type: "
                                                + command.getActionCase())
                                        .build())
                                .build());
                }
            }

            @Override
            public void onCompleted() {
                // Client closed its stream — close ours too
                logger.info("[Trading] ⭐ LiveTrading client stream completed. "
                        + "Active orders: " + activeOrders.size());

                // Send final alert
                responseObserver.onNext(TradeEvent.newBuilder()
                        .setEventId(generateEventId())
                        .setTimestamp(now())
                        .setAlert(SystemAlert.newBuilder()
                                .setLevel(AlertLevel.ALERT_LEVEL_INFO)
                                .setMessage("Session closing. "
                                        + activeOrders.size() + " orders remain active.")
                                .build())
                        .build());

                responseObserver.onCompleted();
            }

            @Override
            public void onError(Throwable t) {
                logger.warning("[Trading] ⭐ LiveTrading stream error: " + t.getMessage());
            }
        };
    }

    // ─── Bidi Handler Methods ────────────────────────

    /**
     * Handle a PLACE_ORDER command.
     * Sends back an ACCEPTED execution, then simulates a FILL after a delay.
     * This demonstrates the server sending MULTIPLE events for a single client command.
     */
    private void handlePlaceOrder(TradeCommand command,
                                   StreamObserver<TradeEvent> responseObserver,
                                   Map<String, TradeOrder> activeOrders) {

        TradeOrder order = command.getPlaceOrder();
        activeOrders.put(order.getOrderId(), order);

        // Event 1: Order accepted
        ExecutionReport accepted = ExecutionReport.newBuilder()
                .setExecutionId(generateExecId())
                .setOrderId(order.getOrderId())
                .setClientId(order.getClientId())
                .setStatus(ExecutionStatus.EXECUTION_STATUS_ACCEPTED)
                .setRemainingQuantity(order.getQuantity())
                .setExecutedAt(now())
                .setMessage("Order accepted and queued for execution")
                .build();

        responseObserver.onNext(TradeEvent.newBuilder()
                .setEventId(generateEventId())
                .setTimestamp(now())
                .setExecution(accepted)
                .build());

        // Simulate async execution — partial fill then full fill
        // In production this would be on a separate thread
        simulateExecution(order, responseObserver, activeOrders);
    }

    /**
     * Simulates order execution with partial fills.
     * Demonstrates server sending multiple events for one client command.
     */
    private void simulateExecution(TradeOrder order,
                                    StreamObserver<TradeEvent> responseObserver,
                                    Map<String, TradeOrder> activeOrders) {

        long totalQty = order.getQuantity().getUnscaledValue();
        long filledSoFar = 0;
        int basePrice = 15000 + random.nextInt(500);

        // Simulate 1-3 partial fills
        int fills = 1 + random.nextInt(3);
        for (int i = 0; i < fills; i++) {
            long fillQty;
            ExecutionStatus status;

            if (i == fills - 1) {
                // Last fill — fill the remainder
                fillQty = totalQty - filledSoFar;
                status = ExecutionStatus.EXECUTION_STATUS_FILLED;
            } else {
                // Partial fill
                fillQty = Math.max(1, (totalQty - filledSoFar) / (fills - i));
                status = ExecutionStatus.EXECUTION_STATUS_PARTIAL_FILL;
            }

            filledSoFar += fillQty;
            int fillPrice = basePrice + random.nextInt(20) - 10;

            ExecutionReport fill = ExecutionReport.newBuilder()
                    .setExecutionId(generateExecId())
                    .setOrderId(order.getOrderId())
                    .setClientId(order.getClientId())
                    .setStatus(status)
                    .setFilledQuantity(makeDecimal(fillQty, 0))
                    .setFillPrice(makeDecimal(fillPrice, 2))
                    .setRemainingQuantity(makeDecimal(totalQty - filledSoFar, 0))
                    .setExecutedAt(now())
                    .setMessage(status == ExecutionStatus.EXECUTION_STATUS_FILLED
                            ? "Order fully filled"
                            : "Partial fill " + (i + 1) + "/" + fills)
                    .build();

            responseObserver.onNext(TradeEvent.newBuilder()
                    .setEventId(generateEventId())
                    .setTimestamp(now())
                    .setExecution(fill)
                    .build());
        }

        // Remove from active orders once fully filled
        activeOrders.remove(order.getOrderId());
    }

    private void handleCancelOrder(TradeCommand command,
                                    StreamObserver<TradeEvent> responseObserver,
                                    Map<String, TradeOrder> activeOrders) {

        CancelOrderRequest cancel = command.getCancelOrder();
        String orderId = cancel.getOrderId();

        if (activeOrders.containsKey(orderId)) {
            activeOrders.remove(orderId);

            responseObserver.onNext(TradeEvent.newBuilder()
                    .setEventId(generateEventId())
                    .setTimestamp(now())
                    .setExecution(ExecutionReport.newBuilder()
                            .setExecutionId(generateExecId())
                            .setOrderId(orderId)
                            .setStatus(ExecutionStatus.EXECUTION_STATUS_CANCELLED)
                            .setExecutedAt(now())
                            .setMessage("Order cancelled: " + cancel.getReason())
                            .build())
                    .build());
        } else {
            responseObserver.onNext(TradeEvent.newBuilder()
                    .setEventId(generateEventId())
                    .setTimestamp(now())
                    .setError(TradeError.newBuilder()
                            .setCommandId(command.getCommandId())
                            .setErrorCode(404)
                            .setMessage("Order not found: " + orderId)
                            .build())
                    .build());
        }
    }

    private void handleModifyOrder(TradeCommand command,
                                    StreamObserver<TradeEvent> responseObserver,
                                    Map<String, TradeOrder> activeOrders) {

        ModifyOrderRequest modify = command.getModifyOrder();
        String orderId = modify.getOrderId();

        if (activeOrders.containsKey(orderId)) {
            responseObserver.onNext(TradeEvent.newBuilder()
                    .setEventId(generateEventId())
                    .setTimestamp(now())
                    .setExecution(ExecutionReport.newBuilder()
                            .setExecutionId(generateExecId())
                            .setOrderId(orderId)
                            .setStatus(ExecutionStatus.EXECUTION_STATUS_ACCEPTED)
                            .setExecutedAt(now())
                            .setMessage("Order modified successfully")
                            .build())
                    .build());
        } else {
            responseObserver.onNext(TradeEvent.newBuilder()
                    .setEventId(generateEventId())
                    .setTimestamp(now())
                    .setError(TradeError.newBuilder()
                            .setCommandId(command.getCommandId())
                            .setErrorCode(404)
                            .setMessage("Cannot modify — order not found: " + orderId)
                            .build())
                    .build());
        }
    }

    private void handleHeartbeat(TradeCommand command,
                                  StreamObserver<TradeEvent> responseObserver,
                                  Map<String, TradeOrder> activeOrders) {

        responseObserver.onNext(TradeEvent.newBuilder()
                .setEventId(generateEventId())
                .setTimestamp(now())
                .setHeartbeat(HeartbeatResponse.newBuilder()
                        .setServerTimeMs(System.currentTimeMillis())
                        .setActiveOrders(activeOrders.size())
                        .build())
                .build());
    }

    // ─── Shared Helpers ──────────────────────────────

    private ExecutionReport executeOrder(TradeOrder order) {
        // Simulate: 90% accepted, 10% rejected
        boolean accepted = random.nextInt(10) > 0;

        if (accepted) {
            int fillPrice = 15000 + random.nextInt(500);
            return ExecutionReport.newBuilder()
                    .setExecutionId(generateExecId())
                    .setOrderId(order.getOrderId())
                    .setClientId(order.getClientId())
                    .setStatus(ExecutionStatus.EXECUTION_STATUS_FILLED)
                    .setFilledQuantity(order.getQuantity())
                    .setFillPrice(makeDecimal(fillPrice, 2))
                    .setRemainingQuantity(makeDecimal(0, 0))
                    .setExecutedAt(now())
                    .setMessage("Order filled at market price")
                    .build();
        } else {
            return ExecutionReport.newBuilder()
                    .setExecutionId(generateExecId())
                    .setOrderId(order.getOrderId())
                    .setClientId(order.getClientId())
                    .setStatus(ExecutionStatus.EXECUTION_STATUS_REJECTED)
                    .setExecutedAt(now())
                    .setMessage("Insufficient margin")
                    .build();
        }
    }

    private Decimal makeDecimal(long unscaled, int scale) {
        return Decimal.newBuilder().setUnscaledValue(unscaled).setScale(scale).build();
    }

    private Timestamp now() {
        long ms = System.currentTimeMillis();
        return Timestamp.newBuilder().setSeconds(ms / 1000)
                .setNanos((int) ((ms % 1000) * 1_000_000)).build();
    }

    private String generateExecId() {
        return "EXEC-" + executionCounter.incrementAndGet();
    }

    private String generateEventId() {
        return "EVT-" + System.nanoTime();
    }
}
```

**Key patterns in the bidi implementation:**
- **Server returns a `StreamObserver`** — this is the handler for incoming client messages
- **`onNext(TradeCommand)`** — fires for each client message; server responds immediately
- **Multiple server events per client command** — `handlePlaceOrder` sends ACCEPTED, then partial fills, then FILLED
- **`oneof` dispatching** — `command.getActionCase()` determines which handler to call
- **Session state** — `activeOrders` map tracks state across the bidirectional stream
- **`onCompleted()`** — server sends a final alert, then closes its stream

## Client Wrappers

Client wrappers provide **clean, type-safe abstractions** over the raw gRPC stubs. Each wrapper manages the channel, stub lifecycle, and `StreamObserver` plumbing so calling code stays simple.

### `MarketDataClient.java`

Wraps the **unary** and **server streaming** calls.

```java
package com.example.trading.client;

import com.example.trading.proto.market.*;

import io.grpc.ManagedChannelBuilder;

import java.util.Arrays;
import java.util.Iterator;
import java.util.concurrent.TimeUnit;
import java.util.function.Consumer;
import java.util.logging.Logger;

/**
 * Client wrapper for MarketDataService.
 * Provides clean methods for market data operations.
 */
public class MarketDataClient implements AutoCloseable {

    private static final Logger logger = Logger.getLogger(MarketDataClient.class.getName());

    private final ManagedChannel channel;
    private final MarketDataServiceGrpc.MarketDataServiceBlockingStub blockingStub;

    // ─── Constructor ─────────────────────────────────

    public MarketDataClient(String host, int port) {
        this.channel = ManagedChannelBuilder
                .forAddress(host, port)
                .usePlaintext()
                .build();
        this.blockingStub = MarketDataServiceGrpc.newBlockingStub(channel);
        logger.info("[MarketDataClient] Connected to " + host + ":" + port);
    }

    // ─── Unary: Get Market Summary ───────────────────

    /**
     * Get the current market summary for a symbol.
     */
    public MarketSummary getMarketSummary(String symbol) {
        MarketSummaryRequest request = MarketSummaryRequest.newBuilder()
                .setSymbol(symbol)
                .build();

        return blockingStub.getMarketSummary(request);
    }

    // ─── Server Streaming: Subscribe to Prices ───────

    /**
     * Subscribe to live price ticks. This blocks the calling thread and invokes
     * the callback for each tick. Use in a separate thread for non-blocking behavior.
     *
     * @param symbols    symbols to subscribe to
     * @param throttleMs minimum interval between ticks
     * @param onTick     callback invoked for each price tick
     */
    public void streamPrices(String[] symbols, int throttleMs, Consumer<PriceTick> onTick) {
        SubscribeRequest request = SubscribeRequest.newBuilder()
                .addAllSymbols(Arrays.asList(symbols))
                .setThrottleMs(throttleMs)
                .build();

        // Blocking iterator — each call to next() blocks until
        // the server sends the next tick
        Iterator<PriceTick> stream = blockingStub.streamPrices(request);

        while (stream.hasNext()) {
            PriceTick tick = stream.next();
            onTick.accept(tick);
        }
    }

    // ─── Lifecycle ───────────────────────────────────

    @Override
    public void close() throws Exception {
        channel.shutdown().awaitTermination(5, TimeUnit.SECONDS);
        logger.info("[MarketDataClient] Channel closed");
    }
}
```

**Wrapper pattern:**
- Constructor manages `ManagedChannel` creation
- Public methods accept simple parameters, build protobuf messages internally
- `streamPrices` uses a `Consumer<PriceTick>` callback — isolates the calling code from `StreamObserver` complexity
- Implements `AutoCloseable` for try-with-resources support

### `TradingClient.java`

Wraps **unary** (`PlaceOrder`) and **client streaming** (`BatchPlaceOrders`).

```java
package com.example.trading.client;

import com.example.trading.proto.common.Decimal;
import com.example.trading.proto.common.Timestamp;
import com.example.trading.proto.market.Instrument;
import com.example.trading.proto.market.InstrumentType;
import com.example.trading.proto.trading.*;
import io.grpc.ManagedChannel;
import io.grpc.ManagedChannelBuilder;
import io.grpc.stub.StreamObserver;

import java.util.List;
import java.util.concurrent.CountDownLatch;
import java.util.concurrent.TimeUnit;
import java.util.concurrent.atomic.AtomicReference;
import java.util.logging.Logger;

/**
 * Client wrapper for TradingService — unary and client streaming operations.
 */
public class TradingClient implements AutoCloseable {

    private static final Logger logger = Logger.getLogger(TradingClient.class.getName());

    private final ManagedChannel channel;
    private final TradingServiceGrpc.TradingServiceBlockingStub blockingStub;
    private final TradingServiceGrpc.TradingServiceStub asyncStub;  // needed for streaming

    // ─── Constructor ─────────────────────────────────

    public TradingClient(String host, int port) {
        this.channel = ManagedChannelBuilder
                .forAddress(host, port)
                .usePlaintext()
                .build();
        this.blockingStub = TradingServiceGrpc.newBlockingStub(channel);
        this.asyncStub = TradingServiceGrpc.newStub(channel);  // async stub for streaming
        logger.info("[TradingClient] Connected to " + host + ":" + port);
    }

    // ─── Unary: Place a single order ─────────────────

    /**
     * Place a single order and wait for the execution report.
     */
    public ExecutionReport placeOrder(String orderId, String clientId,
                                       String symbol, OrderSide side,
                                       OrderType type, long quantity) {

        TradeOrder order = buildOrder(orderId, clientId, symbol, side, type, quantity, 0);

        PlaceOrderRequest request = PlaceOrderRequest.newBuilder()
                .setOrder(order)
                .build();

        return blockingStub.placeOrder(request);
    }

    // ─── Client Streaming: Batch place orders ────────

    /**
     * Send a batch of orders via client streaming.
     * Returns the BatchOrderResponse after all orders are sent and processed.
     *
     * This method demonstrates the key challenge of client streaming:
     * we need the ASYNC stub because we're sending multiple messages,
     * but we want to block until the server responds.
     * CountDownLatch bridges the async/sync gap.
     */
    public BatchOrderResponse batchPlaceOrders(List<TradeOrder> orders)
            throws InterruptedException {

        // Latch to wait for the server's response
        CountDownLatch latch = new CountDownLatch(1);
        AtomicReference<BatchOrderResponse> responseRef = new AtomicReference<>();
        AtomicReference<Throwable> errorRef = new AtomicReference<>();

        // Create a response observer that captures the server's single response
        StreamObserver<BatchOrderResponse> responseObserver =
                new StreamObserver<BatchOrderResponse>() {

            @Override
            public void onNext(BatchOrderResponse response) {
                responseRef.set(response);  // capture the response
            }

            @Override
            public void onCompleted() {
                latch.countDown();  // signal that we're done
            }

            @Override
            public void onError(Throwable t) {
                errorRef.set(t);
                latch.countDown();
            }
        };

        // Open the client stream
        StreamObserver<TradeOrder> requestStream =
                asyncStub.batchPlaceOrders(responseObserver);

        // Send each order
        for (TradeOrder order : orders) {
            logger.info("[TradingClient] Batch sending: " + order.getOrderId());
            requestStream.onNext(order);
        }

        // Signal that we're done sending
        requestStream.onCompleted();

        // Wait for the server to respond (max 30 seconds)
        if (!latch.await(30, TimeUnit.SECONDS)) {
            throw new RuntimeException("Batch order timed out");
        }

        if (errorRef.get() != null) {
            throw new RuntimeException("Batch order failed", errorRef.get());
        }

        return responseRef.get();
    }

    // ─── Helper: Build a TradeOrder ──────────────────

    public static TradeOrder buildOrder(String orderId, String clientId,
                                         String symbol, OrderSide side,
                                         OrderType type, long quantity,
                                         long limitPrice) {

        TradeOrder.Builder builder = TradeOrder.newBuilder()
                .setOrderId(orderId)
                .setClientId(clientId)
                .setInstrument(Instrument.newBuilder()
                        .setSymbol(symbol)
                        .setName(symbol)
                        .setType(InstrumentType.INSTRUMENT_TYPE_STOCK)
                        .build())
                .setSide(side)
                .setType(type)
                .setQuantity(Decimal.newBuilder()
                        .setUnscaledValue(quantity)
                        .setScale(0)
                        .build())
                .setSubmittedAt(Timestamp.newBuilder()
                        .setSeconds(System.currentTimeMillis() / 1000)
                        .build());

        if (type == OrderType.ORDER_TYPE_LIMIT && limitPrice > 0) {
            builder.setLimitPrice(Decimal.newBuilder()
                    .setUnscaledValue(limitPrice)
                    .setScale(2)
                    .build());
        }

        return builder.build();
    }

    // ─── Lifecycle ───────────────────────────────────

    @Override
    public void close() throws Exception {
        channel.shutdown().awaitTermination(5, TimeUnit.SECONDS);
        logger.info("[TradingClient] Channel closed");
    }
}
```

**Key patterns:**
- Both **blocking** and **async** stubs are created — blocking for unary, async for streaming
- `batchPlaceOrders` uses `CountDownLatch` to bridge async `StreamObserver` with synchronous calling code
- `AtomicReference` safely captures the response from the async callback
- `buildOrder` is a static helper — reusable by other clients

### `LiveTradingClient.java`

⭐ The **bidirectional streaming** wrapper — the most complex and important piece.

```java
package com.example.trading.client;

import com.example.trading.proto.common.Decimal;
import com.example.trading.proto.common.Timestamp;
import com.example.trading.proto.trading.*;
import io.grpc.ManagedChannel;
import io.grpc.ManagedChannelBuilder;
import io.grpc.stub.StreamObserver;

import java.util.concurrent.CountDownLatch;
import java.util.concurrent.TimeUnit;
import java.util.concurrent.atomic.AtomicBoolean;
import java.util.function.Consumer;
import java.util.logging.Logger;

/**
 * ⭐ Client wrapper for bidirectional streaming — LiveTrading RPC.
 *
 * This wrapper manages the full lifecycle of a bidi stream:
 * - Opening the stream with an event handler
 * - Sending commands (orders, cancels, modifies, heartbeats)
 * - Receiving events asynchronously via callbacks
 * - Gracefully closing the stream
 *
 * Usage:
 *   LiveTradingClient client = new LiveTradingClient("localhost", 6565);
 *   client.startSession(event -> {
 *       System.out.println("Received: " + event);
 *   });
 *   client.placeOrder(...);
 *   client.cancelOrder(...);
 *   client.endSession();
 *   client.close();
 */
public class LiveTradingClient implements AutoCloseable {

    private static final Logger logger = Logger.getLogger(LiveTradingClient.class.getName());

    private final ManagedChannel channel;
    private final TradingServiceGrpc.TradingServiceStub asyncStub;

    // Stream state
    private StreamObserver<TradeCommand> requestStream;
    private CountDownLatch sessionLatch;
    private final AtomicBoolean sessionActive = new AtomicBoolean(false);
    private int commandCounter = 0;

    // ─── Constructor ─────────────────────────────────

    public LiveTradingClient(String host, int port) {
        this.channel = ManagedChannelBuilder
                .forAddress(host, port)
                .usePlaintext()
                .build();
        // ⭐ Bidirectional streaming REQUIRES the async stub
        this.asyncStub = TradingServiceGrpc.newStub(channel);
        logger.info("[LiveTradingClient] Connected to " + host + ":" + port);
    }

    // ═══════════════════════════════════════════════════
    // SESSION LIFECYCLE
    // ═══════════════════════════════════════════════════

    /**
     * Open a bidirectional streaming session.
     *
     * @param onEvent  callback invoked for each TradeEvent from the server.
     *                 This runs on gRPC's executor thread — keep it fast.
     * @param onError  callback invoked if the stream encounters an error
     * @param onClose  callback invoked when the server closes its stream
     */
    public void startSession(Consumer<TradeEvent> onEvent,
                              Consumer<Throwable> onError,
                              Runnable onClose) {

        if (sessionActive.get()) {
            throw new IllegalStateException("Session already active");
        }

        sessionLatch = new CountDownLatch(1);

        // Create the response observer — handles incoming server events
        StreamObserver<TradeEvent> responseObserver = new StreamObserver<TradeEvent>() {

            @Override
            public void onNext(TradeEvent event) {
                // ⭐ Called asynchronously each time the server sends an event
                onEvent.accept(event);
            }

            @Override
            public void onCompleted() {
                // Server closed its stream
                logger.info("[LiveTradingClient] Server closed the stream");
                sessionActive.set(false);
                onClose.run();
                sessionLatch.countDown();
            }

            @Override
            public void onError(Throwable t) {
                logger.warning("[LiveTradingClient] Stream error: " + t.getMessage());
                sessionActive.set(false);
                onError.accept(t);
                sessionLatch.countDown();
            }
        };

        // ⭐ Open the bidirectional stream — this returns our sending channel
        this.requestStream = asyncStub.liveTrading(responseObserver);
        this.sessionActive.set(true);

        logger.info("[LiveTradingClient] ⭐ Bidi session started");
    }

    /**
     * Convenience overload with default error/close handlers.
     */
    public void startSession(Consumer<TradeEvent> onEvent) {
        startSession(
                onEvent,
                error -> logger.warning("Stream error: " + error.getMessage()),
                () -> logger.info("Session ended")
        );
    }

    /**
     * Close the client's stream. The server may still send events before
     * closing its own stream.
     */
    public void endSession() {
        if (sessionActive.get() && requestStream != null) {
            logger.info("[LiveTradingClient] Closing client stream...");
            requestStream.onCompleted();
        }
    }

    /**
     * Wait for the server to close its stream (i.e., the session to fully end).
     */
    public boolean awaitSessionEnd(long timeout, TimeUnit unit)
            throws InterruptedException {
        if (sessionLatch != null) {
            return sessionLatch.await(timeout, unit);
        }
        return true;
    }

    // ═══════════════════════════════════════════════════
    // COMMAND METHODS — clean API over raw protobuf
    // ═══════════════════════════════════════════════════

    /**
     * Send a PlaceOrder command through the bidi stream.
     */
    public String placeOrder(String orderId, String clientId,
                              String symbol, OrderSide side,
                              OrderType type, long quantity) {

        checkSession();
        TradeOrder order = TradingClient.buildOrder(
                orderId, clientId, symbol, side, type, quantity, 0);

        String commandId = nextCommandId();
        TradeCommand command = TradeCommand.newBuilder()
                .setCommandId(commandId)
                .setClientId(clientId)
                .setTimestamp(now())
                .setPlaceOrder(order)
                .build();

        requestStream.onNext(command);
        logger.info("[LiveTradingClient] → Sent PlaceOrder: " + orderId);
        return commandId;
    }

    /**
     * Send a CancelOrder command through the bidi stream.
     */
    public String cancelOrder(String orderId, String clientId, String reason) {
        checkSession();

        String commandId = nextCommandId();
        TradeCommand command = TradeCommand.newBuilder()
                .setCommandId(commandId)
                .setClientId(clientId)
                .setTimestamp(now())
                .setCancelOrder(CancelOrderRequest.newBuilder()
                        .setOrderId(orderId)
                        .setReason(reason)
                        .build())
                .build();

        requestStream.onNext(command);
        logger.info("[LiveTradingClient] → Sent CancelOrder: " + orderId);
        return commandId;
    }

    /**
     * Send a ModifyOrder command through the bidi stream.
     */
    public String modifyOrder(String orderId, String clientId,
                               long newQuantity, long newLimitPrice) {
        checkSession();

        String commandId = nextCommandId();
        TradeCommand command = TradeCommand.newBuilder()
                .setCommandId(commandId)
                .setClientId(clientId)
                .setTimestamp(now())
                .setModifyOrder(ModifyOrderRequest.newBuilder()
                        .setOrderId(orderId)
                        .setNewQuantity(Decimal.newBuilder()
                                .setUnscaledValue(newQuantity).setScale(0).build())
                        .setNewLimitPrice(Decimal.newBuilder()
                                .setUnscaledValue(newLimitPrice).setScale(2).build())
                        .build())
                .build();

        requestStream.onNext(command);
        logger.info("[LiveTradingClient] → Sent ModifyOrder: " + orderId);
        return commandId;
    }

    /**
     * Send a heartbeat to keep the connection alive.
     */
    public String sendHeartbeat(String clientId) {
        checkSession();

        String commandId = nextCommandId();
        TradeCommand command = TradeCommand.newBuilder()
                .setCommandId(commandId)
                .setClientId(clientId)
                .setTimestamp(now())
                .setHeartbeat(HeartbeatRequest.getDefaultInstance())
                .build();

        requestStream.onNext(command);
        return commandId;
    }

    // ─── Internal Helpers ────────────────────────────

    private void checkSession() {
        if (!sessionActive.get()) {
            throw new IllegalStateException(
                    "No active session. Call startSession() first.");
        }
    }

    private String nextCommandId() {
        return "CMD-" + (++commandCounter);
    }

    private Timestamp now() {
        long ms = System.currentTimeMillis();
        return Timestamp.newBuilder()
                .setSeconds(ms / 1000)
                .setNanos((int) ((ms % 1000) * 1_000_000))
                .build();
    }

    // ─── Lifecycle ───────────────────────────────────

    @Override
    public void close() throws Exception {
        if (sessionActive.get()) {
            endSession();
            awaitSessionEnd(5, TimeUnit.SECONDS);
        }
        channel.shutdown().awaitTermination(5, TimeUnit.SECONDS);
        logger.info("[LiveTradingClient] Channel closed");
    }
}
```

**⭐ Key design decisions in the bidi wrapper:**

| Pattern | Why |
|---------|-----|
| **`startSession()` accepts callbacks** | Events arrive asynchronously on gRPC's thread — callbacks let the caller handle them |
| **`Consumer<TradeEvent>` for events** | Clean functional interface — no raw `StreamObserver` exposed to callers |
| **`requestStream` stored as field** | Lets methods like `placeOrder()` send messages at any time after session starts |
| **`CountDownLatch` for session end** | Lets callers wait for the full session to complete |
| **`AtomicBoolean sessionActive`** | Thread-safe session state — prevents sending after close |
| **`checkSession()` guard** | Throws early if caller tries to send without an active session |
| **`AutoCloseable`** | Ensures graceful cleanup in try-with-resources |
| **Command IDs returned** | Callers can correlate commands with server events |

```java
        // ─────────────────────────────────────────────
        // Demo 4: ⭐ Bidirectional Streaming — LiveTrading
        // ─────────────────────────────────────────────
        System.out.println("━━━ DEMO 4: ⭐ Bidirectional Streaming — LiveTrading ━━━\n");

        try (LiveTradingClient liveClient = new LiveTradingClient(HOST, PORT)) {

            // Start the bidi session with an event handler
            liveClient.startSession(
                    // onEvent — handle each incoming TradeEvent
                    event -> {
                        switch (event.getPayloadCase()) {
                            case EXECUTION:
                                ExecutionReport exec = event.getExecution();
                                System.out.printf("  ◄ EXECUTION [%s] %s — %s (filled: %s @ %s)%n",
                                        exec.getOrderId(),
                                        exec.getStatus(),
                                        exec.getMessage(),
                                        formatDecimal(exec.getFilledQuantity()),
                                        formatDecimal(exec.getFillPrice()));
                                break;

                            case ERROR:
                                TradeError err = event.getError();
                                System.out.printf("  ◄ ERROR [cmd=%s] code=%d — %s%n",
                                        err.getCommandId(), err.getErrorCode(),
                                        err.getMessage());
                                break;

                            case HEARTBEAT:
                                HeartbeatResponse hb = event.getHeartbeat();
                                System.out.printf("  ◄ HEARTBEAT server_time=%d active_orders=%d%n",
                                        hb.getServerTimeMs(), hb.getActiveOrders());
                                break;

                            case ALERT:
                                SystemAlert alert = event.getAlert();
                                System.out.printf("  ◄ ALERT [%s] %s%n",
                                        alert.getLevel(), alert.getMessage());
                                break;

                            default:
                                System.out.println("  ◄ Unknown event: " + event);
                        }
                    },

                    // onError
                    error -> System.err.println("  ✗ Stream error: " + error.getMessage()),

                    // onClose
                    () -> System.out.println("  ✓ Server closed the session")
            );

            // Give the welcome alert time to arrive
            Thread.sleep(500);

            // ── Send commands through the bidi stream ──

            System.out.println("\n  ► Placing order LIVE-001 (BUY 150 AAPL)...");
            liveClient.placeOrder("LIVE-001", "trader-1", "AAPL",
                    OrderSide.ORDER_SIDE_BUY, OrderType.ORDER_TYPE_MARKET, 150);
            Thread.sleep(300);

            System.out.println("  ► Placing order LIVE-002 (SELL 75 GOOGL)...");
            liveClient.placeOrder("LIVE-002", "trader-1", "GOOGL",
                    OrderSide.ORDER_SIDE_SELL, OrderType.ORDER_TYPE_MARKET, 75);
            Thread.sleep(300);

            System.out.println("  ► Sending heartbeat...");
            liveClient.sendHeartbeat("trader-1");
            Thread.sleep(300);

            System.out.println("  ► Cancelling order LIVE-001...");
            liveClient.cancelOrder("LIVE-001", "trader-1", "Changed my mind");
            Thread.sleep(300);

            System.out.println("  ► Modifying non-existent order LIVE-999...");
            liveClient.modifyOrder("LIVE-999", "trader-1", 500, 20000);
            Thread.sleep(300);

            System.out.println("  ► Placing order LIVE-003 (BUY 200 MSFT)...");
            liveClient.placeOrder("LIVE-003", "trader-1", "MSFT",
                    OrderSide.ORDER_SIDE_BUY, OrderType.ORDER_TYPE_LIMIT, 200);
            Thread.sleep(500);

            // Close the session
            System.out.println("\n  ► Ending live trading session...");
            liveClient.endSession();
            liveClient.awaitSessionEnd(10, TimeUnit.SECONDS);
        }

        System.out.println("\n═══════════════════════════════════════════");
        System.out.println("  All demos completed!");
        System.out.println("═══════════════════════════════════════════");
    }

    // ─── Helper ──────────────────────────────────────

    private static String formatDecimal(
            com.example.trading.proto.common.Decimal d) {
        if (d == null || d.getScale() == 0) {
            return String.valueOf(d == null ? 0 : d.getUnscaledValue());
        }
        double value = d.getUnscaledValue() / Math.pow(10, d.getScale());
        return String.format("%.2f", value);
    }
}
```

## Summary — All Four Patterns Implemented

| Pattern | Proto RPC | Server Class | Client Wrapper | Key Java Types |
|---------|-----------|-------------|----------------|----------------|
| **Unary** | `PlaceOrder` / `GetMarketSummary` | Direct `responseObserver.onNext()` | `blockingStub.method()` | `BlockingStub` |
| **Server Streaming** | `StreamPrices` | Loop + `onNext()` + `onCompleted()` | `Iterator<T>` from blocking stub | `BlockingStub` |
| **Client Streaming** | `BatchPlaceOrders` | Return `StreamObserver<T>` | `asyncStub` + `CountDownLatch` | `Stub` (async) |
| **Bidirectional** | `LiveTrading` | Return `StreamObserver<T>` + send via captured `responseObserver` | `asyncStub` + callbacks + session lifecycle | `Stub` (async) |

### Client Wrapper Design Patterns

| Pattern | Purpose |
|---------|---------|
| **`AutoCloseable`** | Ensures channel cleanup in try-with-resources |
| **`Consumer<T>` callbacks** | Shields callers from raw `StreamObserver` |
| **`CountDownLatch`** | Bridges async gRPC with synchronous calling code |
| **`AtomicReference`** | Thread-safe capture of async results |
| **`AtomicBoolean` session state** | Prevents sending on closed streams |
| **Builder helpers** | Static methods to construct protobuf messages cleanly |
| **Command ID tracking** | Enables correlation of commands with server events |
